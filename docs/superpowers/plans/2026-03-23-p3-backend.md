```python
def check_reds_flags(days_back=180):
    """Screen for RED-S (Relative Energy Deficiency in Sport) risk factors.

    Checks:
    - Performance declining while training load maintained
    - Illness frequency: 5+ consecutive days off, excluding rest weeks (prior week TSS > 200)

    Source: Persp-36 (Traci Carson), Stellingwerff.
    """
    activities = get_activities()
    if activities.empty or len(activities) < 10:
        return None

    cutoff = (pd.Timestamp.now() - pd.Timedelta(days=days_back)).strftime("%Y-%m-%d")
    recent = activities[activities["start_time"] >= cutoff].sort_values("start_time")

    if len(recent) < 5:
        return None

    flags = []

    # 1. Detect illness gaps (5+ consecutive days, not rest weeks)
    dates = pd.to_datetime(recent["start_time"]).dt.date.unique()
    dates = sorted(dates)

    if len(dates) >= 2:
        for i in range(1, len(dates)):
            gap_days = (dates[i] - dates[i - 1]).days
            if gap_days >= 5:
                # Check if prior week had high TSS (rest week exclusion)
                gap_start = pd.Timestamp(dates[i - 1])
                prior_week = activities[
                    (activities["start_time"] >= (gap_start - pd.Timedelta(days=7)).strftime("%Y-%m-%d")) &
                    (activities["start_time"] < gap_start.strftime("%Y-%m-%d"))
                ]
                prior_tss = 0
                ftp = get_config().get("ftp") or 292
                for _, act in prior_week.iterrows():
                    np_val = act.get("normalized_power")
                    dur = act.get("total_timer_time", 0)
                    if np_val and np_val > 0 and dur > 0:
                        prior_tss += (dur * np_val ** 2) / (ftp ** 2 * 3600) * 100

                if prior_tss <= 200:  # NOT a rest week — likely illness
                    flags.append({
                        "type": "illness_gap",
                        "gap_days": gap_days,
                        "date": str(dates[i - 1]),
                        "message": f"{gap_days}-day gap starting {dates[i-1]} (not a rest week)",
                    })

    # 2. Check for performance decline with maintained load
    # Compare last 30 days avg NP to prior 30-60 days
    now = pd.Timestamp.now()
    last_30 = recent[recent["start_time"] >= (now - pd.Timedelta(days=30)).strftime("%Y-%m-%d")]
    prior_30 = recent[
        (recent["start_time"] >= (now - pd.Timedelta(days=60)).strftime("%Y-%m-%d")) &
        (recent["start_time"] < (now - pd.Timedelta(days=30)).strftime("%Y-%m-%d"))
    ]

    if len(last_30) >= 3 and len(prior_30) >= 3:
        recent_np = last_30["normalized_power"].dropna().mean()
        prior_np = prior_30["normalized_power"].dropna().mean()
        if prior_np > 0 and recent_np < prior_np * 0.95:
            # Check if volume was maintained
            recent_vol = last_30["total_timer_time"].sum()
            prior_vol = prior_30["total_timer_time"].sum()
            if recent_vol >= prior_vol * 0.85:
                flags.append({
                    "type": "performance_decline",
                    "recent_np": round(float(recent_np), 1),
                    "prior_np": round(float(prior_np), 1),
                    "decline_pct": round((1 - recent_np / prior_np) * 100, 1),
                    "message": f"NP declined {round((1 - recent_np / prior_np) * 100, 1)}% while volume maintained",
                })

    # Determine risk level
    illness_count = len([f for f in flags if f["type"] == "illness_gap"])
    if illness_count >= 2 or any(f["type"] == "performance_decline" for f in flags):
        risk_level = "high"
    elif illness_count >= 1:
        risk_level = "moderate"
    else:
        risk_level = "low"

    return {
        "risk_level": risk_level,
        "flags": flags,
        "recommendation": (
            "Consider underfueling (RED-S) — check energy availability before assuming overtraining."
            if risk_level == "high"
            else "Monitor — track energy intake and recovery quality."
            if risk_level == "moderate"
            else "No RED-S indicators detected."
        ),
    }


def check_within_day_deficit(activity_id):
    """Estimate within-day energy deficit risk.

    Flags when a high-kJ ride ends late or has no follow-up activity
    soon after (suggesting delayed refueling).

    Source: WD-59 (effect size 2.03 for delayed carbs), Persp-36.
    """
    activities = get_activities()
    if activities.empty:
        return None

    activity = activities[activities["id"] == activity_id]
    if activity.empty:
        return None

    act = activity.iloc[0]
    total_kj = act.get("total_work")
    if total_kj is None or total_kj <= 0:
        # Estimate from power and duration
        avg_power = act.get("avg_power", 0) or 0
        duration_s = act.get("total_timer_time", 0) or 0
        total_kj = avg_power * duration_s / 1000

    if total_kj < 500:
        return None  # short/easy ride, no deficit concern

    start_time = act.get("start_time", "")
    duration_s = act.get("total_timer_time", 0) or 0

    try:
        end_time = pd.Timestamp(start_time) + pd.Timedelta(seconds=duration_s)
        end_hour = end_time.hour
    except Exception:
        return None

    # High deficit risk: ride ends after 7pm (delayed dinner/recovery)
    late_finish = end_hour >= 19

    # Check distance to next activity
    later = activities[activities["start_time"] > start_time].sort_values("start_time")
    if not later.empty:
        next_start = pd.Timestamp(later.iloc[0]["start_time"])
        hours_to_next = (next_start - end_time).total_seconds() / 3600
        back_to_back = hours_to_next < 4
    else:
        hours_to_next = None
        back_to_back = False

    deficit_kcal = total_kj / 4.184  # rough kJ to kcal

    risk_factors = []
    if late_finish:
        risk_factors.append("Late finish (after 7pm) — recovery meal likely delayed")
    if back_to_back:
        risk_factors.append(f"Back-to-back ride in {hours_to_next:.1f}h — insufficient recovery window")
    if deficit_kcal > 1500:
        risk_factors.append(f"High energy cost ({deficit_kcal:.0f} kcal) — large within-day deficit likely")

    if not risk_factors:
        return None

    return {
        "deficit_risk": "high" if len(risk_factors) >= 2 else "moderate",
        "total_kj": round(total_kj, 0),
        "deficit_kcal_estimate": round(deficit_kcal, 0),
        "risk_factors": risk_factors,
        "end_hour": end_hour,
        "hours_to_next_activity": round(hours_to_next, 1) if hours_to_next else None,
    }
```

```python
def ftp_growth_curve(window_days=90, step_days=30):
    """Fit logarithmic growth model to rolling FTP history.

    Returns growth rate, plateau detection, and growth phase classification.
    Source: WD-61 (Steel et al., N=14,690) — diminishing returns follow log curve.
    """
    from wko5.pdcurve import rolling_ftp
    from scipy.optimize import curve_fit

    rf = rolling_ftp(window_days=window_days, step_days=step_days)
    if rf.empty or len(rf) < 4:
        return None

    # Convert dates to weeks from first observation
    dates = pd.to_datetime(rf["date"])
    weeks = ((dates - dates.iloc[0]).dt.days / 7).values
    ftp_vals = rf["mFTP"].values

    if len(weeks) < 4 or np.std(ftp_vals) < 1:
        return None

    # Fit log model: FTP = a * ln(weeks + 1) + b
    def log_model(x, a, b):
        return a * np.log(x + 1) + b

    try:
        popt, _ = curve_fit(log_model, weeks, ftp_vals, p0=[10, ftp_vals[0]], maxfev=5000)
        a, b = popt
        y_pred = log_model(weeks, a, b)
        ss_res = np.sum((ftp_vals - y_pred) ** 2)
        ss_tot = np.sum((ftp_vals - np.mean(ftp_vals)) ** 2)
        r_squared = 1 - ss_res / ss_tot if ss_tot > 0 else 0
    except (RuntimeError, ValueError):
        a, b, r_squared = 0, float(np.mean(ftp_vals)), 0

    # Current improvement rate (derivative of log at current week)
    current_week = weeks[-1]
    rate_w_per_week = a / (current_week + 1) if current_week > 0 else a
    rate_w_per_year = rate_w_per_week * 52

    # Classify growth phase
    if rate_w_per_year > 25:
        phase = "early"
    elif rate_w_per_year > 10:
        phase = "intermediate"
    elif rate_w_per_year > 2:
        phase = "mature"
    else:
        phase = "plateau"

    return {
        "slope": round(float(a), 2),
        "intercept": round(float(b), 1),
        "r_squared": round(float(r_squared), 3),
        "improvement_rate_w_per_year": round(float(rate_w_per_year), 1),
        "plateau_detected": rate_w_per_year < 2,
        "growth_phase": phase,
        "training_age_weeks": int(current_week),
        "data_points": len(ftp_vals),
    }


def performance_trend(durations=None, days_back=30):
    """Track best effort at key durations per ride. 7-day rolling trend.

    Source: TMT-73 — day-to-day performance trending down = investigate.
    """
    if durations is None:
        durations = [300, 1200]

    activities = get_activities()
    cutoff = (pd.Timestamp.now() - pd.Timedelta(days=days_back)).strftime("%Y-%m-%d")
    recent = activities[activities["start_time"] >= cutoff].sort_values("start_time")

    if len(recent) < 3:
        return None

    rows = []
    for _, act in recent.iterrows():
        records = get_records(act["id"])
        if records.empty or "power" not in records.columns:
            continue

        power = records["power"].fillna(0).values.astype(float)
        n = len(power)
        if n < min(durations):
            continue

        cumsum = np.concatenate([[0], np.cumsum(power)])
        row = {"date": str(act["start_time"])[:10], "activity_id": int(act["id"])}

        for d in durations:
            if d <= n:
                rolling = (cumsum[d:] - cumsum[:n - d + 1]) / d
                row[f"best_{d}s"] = round(float(rolling.max()), 1)
            else:
                row[f"best_{d}s"] = float("nan")

        rows.append(row)

    if not rows:
        return None

    df = pd.DataFrame(rows)
    return df
```

```python
def opportunity_cost_analysis(route_id):
    """Rank training priorities by event-specific impact.

    Derives race demands from route segments, compares against current
    power profile + durability. Ranks dimensions by estimated benefit.

    Source: TMT-70, TMT-60.
    """
    from wko5.segments import analyze_ride_segments
    from wko5.pdcurve import compute_envelope_mmp, fit_pd_model
    from wko5.durability import fit_durability_model
    from wko5.routes import get_route

    route = get_route(route_id)
    if route is None:
        return None

    # Get route segments from a historical ride on this route
    from wko5.db import get_connection
    conn = get_connection()
    cursor = conn.cursor()
    cursor.execute(
        "SELECT activity_id FROM route_links WHERE route_id = ? ORDER BY activity_id DESC LIMIT 1",
        (route_id,)
    )
    row = cursor.fetchone()
    conn.close()

    if row is None:
        return None

    ride = analyze_ride_segments(row[0])
    if not ride or not ride.get("segments"):
        return None

    segments = ride["segments"]

    # Current power profile
    mmp = compute_envelope_mmp(days=90)
    if len(mmp) < 300:
        return None

    pd_model = fit_pd_model(mmp)
    if pd_model is None:
        return None

    dur_params = fit_durability_model()

    # Compute impact of +10W at different durations
    dimensions = []

    # FTP impact: how much faster with +10W mFTP
    max_demand = max((s.get("power_required", 0) for s in segments), default=0)
    total_duration = sum(s.get("duration_s", 0) for s in segments)
    ftp_impact = 0
    for seg in segments:
        pr = seg.get("power_required", 0)
        dur = seg.get("duration_s", 0)
        if pr > 0 and dur > 0:
            # Rough: 10W more capacity = proportional time savings on hard segments
            ftp_impact += dur * min(10 / max(pr, 1), 0.1)

    dimensions.append({
        "dimension": "FTP (+10W)",
        "impact": round(ftp_impact / 60, 1),  # minutes saved
        "level": "high" if ftp_impact / 60 > 5 else "medium" if ftp_impact / 60 > 2 else "low",
    })

    # Durability impact
    total_kj = sum(s.get("avg_power", 0) * s.get("duration_s", 0) / 1000 for s in segments if s.get("avg_power"))
    dur_impact = total_kj / 1000 * 2  # rough: better durability saves ~2 min per 1000kJ on long routes
    dimensions.append({
        "dimension": "Durability (-2% fade)",
        "impact": round(dur_impact, 1),
        "level": "high" if dur_impact > 5 else "medium" if dur_impact > 2 else "low",
    })

    # VO2max impact
    vo2_segments = [s for s in segments if s.get("power_required", 0) > pd_model.get("mFTP", 280)]
    vo2_impact = sum(s.get("duration_s", 0) for s in vo2_segments) * 0.02
    dimensions.append({
        "dimension": "VO2max (+5W)",
        "impact": round(vo2_impact / 60, 1),
        "level": "high" if vo2_impact / 60 > 3 else "medium" if vo2_impact / 60 > 1 else "low",
    })

    # Sprint impact
    sprint_segs = [s for s in segments if s.get("avg_grade", 0) < -0.02]
    sprint_impact = len(sprint_segs) * 0.2  # minimal for most routes
    dimensions.append({
        "dimension": "Sprint (+50W)",
        "impact": round(sprint_impact, 1),
        "level": "low",
    })

    # Nutrition impact (based on ride duration)
    nutrition_impact = max(0, (total_duration / 3600 - 4) * 1.5)  # grows with duration beyond 4h
    dimensions.append({
        "dimension": "Nutrition optimization",
        "impact": round(nutrition_impact, 1),
        "level": "high" if nutrition_impact > 5 else "medium" if nutrition_impact > 2 else "low",
    })

    # Sort by impact descending
    dimensions.sort(key=lambda x: x["impact"], reverse=True)
    return dimensions


def short_power_consistency(duration_s=60, days_back=365):
    """Check consistency of short-duration power across the year.

    Compares peak vs median effort. Ratio > 1.3 = consistency problem.
    Source: TMT-64.
    """
    from wko5.db import get_activities, get_records

    activities = get_activities()
    cutoff = (pd.Timestamp.now() - pd.Timedelta(days=days_back)).strftime("%Y-%m-%d")
    recent = activities[activities["start_time"] >= cutoff]

    efforts = []
    for _, act in recent.iterrows():
        records = get_records(act["id"])
        if records.empty or "power" not in records.columns:
            continue

        power = records["power"].fillna(0).values.astype(float)
        n = len(power)
        if n < duration_s:
            continue

        cumsum = np.concatenate([[0], np.cumsum(power)])
        rolling = (cumsum[duration_s:] - cumsum[:n - duration_s + 1]) / duration_s
        if len(rolling) > 0:
            efforts.append(float(rolling.max()))

    if len(efforts) < 5:
        return None

    peak = max(efforts)
    typical = float(np.median(efforts))
    ratio = peak / typical if typical > 0 else 0

    return {
        "peak": round(peak, 1),
        "typical": round(typical, 1),
        "ratio": round(ratio, 2),
        "diagnosis": "consistency" if ratio > 1.3 else "capacity",
        "efforts_analyzed": len(efforts),
        "message": (
            f"Peak {duration_s}s power ({peak:.0f}W) is {ratio:.1f}x your typical ({typical:.0f}W). "
            f"{'Focus on reproducing peak efforts more consistently.' if ratio > 1.3 else 'Power is consistent — capacity is the limiter.'}"
        ),
    }
```

```python
def check_absorption_ceiling(intake_g_hr, ceiling_g_hr=90):
    """Flag when carb intake exceeds likely absorption ceiling.

    Source: Persp-41 — absorption ranges 50-150g/hr individually.
    Default ceiling 90g/hr unless athlete has tested higher.
    """
    if intake_g_hr <= ceiling_g_hr:
        return None
    return {
        "warning": f"Intake {intake_g_hr}g/hr exceeds absorption ceiling ({ceiling_g_hr}g/hr). "
                   f"Risk of GI distress. Test gut tolerance before racing at this rate.",
        "intake": intake_g_hr,
        "ceiling": ceiling_g_hr,
        "excess": round(intake_g_hr - ceiling_g_hr, 1),

def glycogen_budget_daily(ride_kj, ride_duration_h, on_bike_carbs_g,
                           post_ride_delay_h, daily_carb_target_g_kg, weight_kg):
    """Model daily glycogen budget with recovery timing.

    Source: WD-59 (delayed carbs impair next-day performance by ~30%),
    Persp-41 (120g/hr on-bike can squeeze recovery carb budget).
    """
    # Glycogen capacity: ~500g for 78kg athlete (~6.4 g/kg)
    glycogen_capacity_g = weight_kg * 6.4

    # Energy from ride (kcal)
    ride_kcal = ride_kj / 4.184

    # CHO burned on ride (rough: 50-70% from carbs at typical intensity)
    cho_fraction = 0.60  # approximate for mixed endurance
    cho_burned_g = ride_kcal * cho_fraction / 4  # 4 kcal per g CHO

    # Net glycogen cost = burned - eaten on bike
    net_glycogen_cost_g = max(0, cho_burned_g - on_bike_carbs_g)

    # Recovery carbs available
    daily_carb_total_g = daily_carb_target_g_kg * weight_kg
    recovery_carbs_g = max(0, daily_carb_total_g - on_bike_carbs_g)

    # Recovery rate: 1.2 g/kg/hr optimal (Persp-41)
    optimal_rate_g_hr = 1.2 * weight_kg
    hours_available = max(0, 16 - ride_duration_h - post_ride_delay_h)  # waking hours minus ride + delay
    achievable_repletion_g = min(recovery_carbs_g, optimal_rate_g_hr * hours_available)

    # Next-day glycogen estimate
    starting_glycogen_g = glycogen_capacity_g  # assume full start
    post_ride_g = max(0, starting_glycogen_g - net_glycogen_cost_g)
    next_day_g = min(glycogen_capacity_g, post_ride_g + achievable_repletion_g)
    next_day_pct = round(next_day_g / glycogen_capacity_g * 100, 1)

    warning = None
    if post_ride_delay_h > 2:
        warning = (f"Post-ride refueling delayed {post_ride_delay_h}h — "
                   f"next-day performance may drop ~30% (WD-59).")
    elif next_day_pct < 70:
        warning = (f"Next-day glycogen only {next_day_pct}% — "
                   f"increase daily carbs or reduce post-ride delay.")

    return {
        "cho_burned_g": round(cho_burned_g, 0),
        "on_bike_carbs_g": on_bike_carbs_g,
        "net_glycogen_cost_g": round(net_glycogen_cost_g, 0),
        "recovery_carbs_available_g": round(recovery_carbs_g, 0),
        "recovery_hours_available": round(hours_available, 1),
        "achievable_repletion_g": round(achievable_repletion_g, 0),
        "next_day_glycogen_pct": next_day_pct,
        "warning": warning,
    }
```

```python
def energy_expenditure(power_watts, efficiency=0.23, with_uncertainty=False):
    """Compute energy expenditure in kcal/hr from power output.

    If with_uncertainty=True, returns (mid, low, high) reflecting
    ~900 kcal swing from efficiency assumptions (20-25% GE range).
    Source: Persp-41.
    """
    if power_watts <= 0:
        if with_uncertainty:
            return (80.0, 70.0, 90.0)
        return 80.0

    mid = power_watts * 3.6 / (efficiency * 4.184)

    if not with_uncertainty:
        return mid

    low = power_watts * 3.6 / (0.25 * 4.184)   # higher efficiency = lower EE
    high = power_watts * 3.6 / (0.20 * 4.184)   # lower efficiency = higher EE
    return (mid, low, high)
```

```python
def rolling_pd_profile(window_days=90, step_days=14):
    """Compute full PD model parameters at regular intervals.

    Returns DataFrame with date, mFTP, Pmax, FRC, TTE tracked over time.
    Source: WD-62, TMT-66 — rolling curves more reliable than single tests.
    """
    activities = get_activities()
    if activities.empty:
        return None

    dates = pd.to_datetime(activities["start_time"])
    min_date = dates.min()
    max_date = dates.max()

    results = []
    current = min_date + pd.Timedelta(days=window_days)

    while current <= max_date:
        start = (current - pd.Timedelta(days=window_days)).strftime("%Y-%m-%d")
        end = current.strftime("%Y-%m-%d 23:59:59")

        mmp = compute_envelope_mmp(start=start, end=end)
        if len(mmp) >= 60:
            model = fit_pd_model(mmp)
            if model:
                results.append({
                    "date": current.strftime("%Y-%m-%d"),
                    "mFTP": model.get("mFTP"),
                    "Pmax": model.get("Pmax"),
                    "FRC": model.get("FRC"),
                    "TTE": model.get("TTE"),
                })

        current += pd.Timedelta(days=step_days)

    if not results:
        return None

    return pd.DataFrame(results)
```
